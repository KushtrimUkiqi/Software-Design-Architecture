The data is taken from : <a href="https://gisco-services.ec.europa.eu/pub/healthcare/csv/all.csv"> https://gisco-services.ec.europa.eu/pub/healthcare/csv/all.csv</a>



SwissHospitals <- all %>%
filter(country == "Switzerland") %>%
select(id,city,lon,lat,hospital_name,postcode,street) %>%
rename(
Id = id,
City = city,
Longitude = lon,
Latitude =lat,
Name = hospital_name,
PostCode = postcode,
Street = street
)
write.csv(SwissHospitals, "SwissHospitals.csv")
